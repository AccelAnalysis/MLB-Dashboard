// --- HELPER FUNCTIONS ---
const toISODate = (date) => date.toISOString().split('T')[0];
const todayISO = () => toISODate(new Date());
const daysAgo = (days) => toISODate(new Date(Date.now() - days * 86400000));
const daysFromNow = (days) => toISODate(new Date(Date.now() + days * 86400000));

const currency = (value) => new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD', maximumFractionDigits: 0 }).format(value || 0);
const formatDate = (dateStr) => {
  if (!dateStr) return '-';
  return new Intl.DateTimeFormat('en-US', { month: 'short', day: 'numeric', year: 'numeric' }).format(new Date(`${dateStr}T00:00:00`));
};
const daysBetween = (newerDate, olderDate) => {
  if (!newerDate || !olderDate) return 0;
  return Math.floor((new Date(`${newerDate}T00:00:00`) - new Date(`${olderDate}T00:00:00`)) / 86400000);
};

// --- LOGIC CALCULATIONS ---
const getRevisedAmount = (project) => {
  return project.originalAmount + (project.changeOrders?.reduce((sum, co) => sum + Number(co.amount), 0) || 0);
};

const getScopeStatus = (scope) => {
  if (scope.completionDate) return { label: 'Completed', color: 'bg-gray-200 text-gray-800', stage: 7 };
  if (scope.scheduledInstallDate) return { label: 'Scheduled', color: 'bg-pink-100 text-pink-700 border border-pink-200', stage: 6 };
  if (scope.materialsIn) return { label: 'Materials In', color: 'bg-orange-100 text-orange-700 border border-orange-200', stage: 5 };
  if (scope.materialETA) return { label: 'Material ETA', color: 'bg-blue-100 text-blue-700 border border-blue-200', stage: 4 };
  if (scope.dateOrdered) return { label: 'Ordered', color: 'bg-teal-100 text-teal-700 border border-teal-200', stage: 3 };
  if (scope.materialListReceived) return { label: 'List Rcvd / Ready to Order', color: 'bg-green-100 text-green-800 border border-green-200', stage: 2 };
  if (scope.measureCompleted) return { label: 'Measured / Needs List', color: 'bg-indigo-100 text-indigo-700 border border-indigo-200', stage: 1 };
  if (scope.measurer) return { label: 'Measure Assigned', color: 'bg-purple-100 text-purple-700 border border-purple-200', stage: 0 };
  return { label: 'Needs Measurer', color: 'bg-red-50 text-red-700 border border-red-200', stage: -1 };
};

const calculateNextAction = (scope) => {
  if (scope.completionDate) return "Close out / Collect";
  if (scope.scheduledInstallDate) return "Execute Installation";
  if (scope.materialsIn) return "Schedule Customer & Assign Crew";
  if (scope.materialETA) return "Track Material Delivery";
  if (scope.dateOrdered) return "Confirm Material ETA";
  if (scope.materialListReceived) return "Order Materials";
  if (scope.measureCompleted) return "Submit Material List";
  if (scope.measurer) return "Complete Measurement";
  return "Assign Measurer";
};

          )}

        {/* Footer */}
        <div className="flex justify-end gap-3 border-t border-slate-200 bg-white px-6 py-4">

        </div>

      {/* Nested Scope Editor Modal */}
      {editingScope && (

              </div>

                <h4 className="font-bold text-slate-800 border-b border-slate-200 pb-2">Measurement & Materials</h4>
                <div className="grid grid-cols-2 gap-4">
                  <label className="block text-xs font-bold text-slate-500 uppercase">Measurer<input type="text" value={editingScope.measurer || ''} onChange={e => setEditingScope({...editingScope, measurer: e.target.value})} className="mt-1 w-full rounded border-slate-300 p-2 text-sm" /></label>
                  <label className="block text-xs font-bold text-slate-500 uppercase">Measure Done<input type="date" value={editingScope.measureCompleted || ''} onChange={e => setEditingScope({...editingScope, measureCompleted: e.target.value})} className="mt-1 w-full rounded border-slate-300 p-2 text-sm" /></label>
                  <label className="block text-xs font-bold text-slate-500 uppercase">Mat. List Received<input type="date" value={editingScope.materialListReceived || ''} onChange={e => setEditingScope({...editingScope, materialListReceived: e.target.value})} className="mt-1 w-full rounded border-slate-300 p-2 text-sm" /></label>
                  <label className="block text-xs font-bold text-slate-500 uppercase">Date Ordered<input type="date" value={editingScope.dateOrdered || ''} onChange={e => setEditingScope({...editingScope, dateOrdered: e.target.value})} className="mt-1 w-full rounded border-slate-300 p-2 text-sm" /></label>
                  <label className="block text-xs font-bold text-slate-500 uppercase">Vendor<input type="text" value={editingScope.vendor || ''} onChange={e => setEditingScope({...editingScope, vendor: e.target.value})} className="mt-1 w-full rounded border-slate-300 p-2 text-sm" /></label>
                  <label className="block text-xs font-bold text-slate-500 uppercase">Material ETA<input type="date" value={editingScope.materialETA || ''} onChange={e => setEditingScope({...editingScope, materialETA: e.target.value})} className="mt-1 w-full rounded border-slate-300 p-2 text-sm" /></label>
                </div>
              </div>

              <div className="bg-slate-50 p-4 rounded-xl border border-slate-200 space-y-4">
                <h4 className="font-bold text-slate-800 border-b border-slate-200 pb-2">Production Schedule</h4>
                <div className="grid grid-cols-2 gap-4">
                  <label className="block text-xs font-bold text-slate-500 uppercase">Materials In<input type="date" value={editingScope.materialsIn || ''} onChange={e => setEditingScope({...editingScope, materialsIn: e.target.value})} className="mt-1 w-full rounded border-slate-300 p-2 text-sm" /></label>
                  <label className="block text-xs font-bold text-slate-500 uppercase">Scheduled Install<input type="date" value={editingScope.scheduledInstallDate || ''} onChange={e => setEditingScope({...editingScope, scheduledInstallDate: e.target.value})} className="mt-1 w-full rounded border-slate-300 p-2 text-sm" /></label>
                  <label className="block text-xs font-bold text-slate-500 uppercase">Completion Date<input type="date" value={editingScope.completionDate || ''} onChange={e => setEditingScope({...editingScope, completionDate: e.target.value})} className="mt-1 w-full rounded border-slate-300 p-2 text-sm" /></label>
                </div>
              </div>
            </div>
            <div className="p-4 bg-slate-50 border-t border-slate-200 flex justify-end gap-3">
              <button onClick={() => setEditingScope(null)} className="px-4 py-2 border border-slate-300 rounded-lg text-sm font-bold">Cancel</button>
              <button onClick={() => saveScope(editingScope)} className="px-4 py-2 bg-slate-900 text-white rounded-lg text-sm font-bold">Save Scope</button>
            </div>
          </div>
        </div>
      )}
    </div>
  );
};
